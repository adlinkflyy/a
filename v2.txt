#!/bin/bash

echo "==================== CPU Info ===================="
lscpu | grep -E '^Model name|^CPUs|^Thread|^Core|^Socket'

echo ""
echo "==================== RAM Info ===================="
free -h

echo ""
echo "==================== GPU Info ===================="
if command -v nvidia-smi &> /dev/null; then
    echo "GPU NVIDIA (dùng nvidia-smi):"
    nvidia-smi
else
    echo "GPU hiển thị (dùng lspci):"
    lspci | grep -i vga
    echo ""
    echo "Chi tiết GPU (dùng lshw):"
    if command -v lshw &> /dev/null; then
        sudo lshw -C display | grep -E 'product|vendor|configuration|driver'
    else
        echo "Bạn cần cài đặt lshw: sudo apt install lshw"
    fi
fi
